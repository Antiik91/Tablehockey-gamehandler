197     private void addPlayerButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_addPlayerButtonActionPerformed
198         // Appears to be null for some reason..
199         evt.getSource();
200         if (this.currentStandings != null) {
201             JFrame frame = new JFrame("Add Player to standings");
202             frame.setDefaultCloseOperation(HIDE_ON_CLOSE);
203 
204             JPanel content = new JPanel();
205 
206             content.setBorder(
207                     BorderFactory.createEmptyBorder(5, 5, 5, 5));
208             content.setLayout(new BorderLayout());
209 
210             content.add(new JTextField());
211 
212             JButton playerButton = new JButton("Add Player");
213             playerButton.addActionListener(new AddPlayerListener(this.currentStandingTextField, currentStandings));
214             content.add(playerButton);
215 
216             JTextField newPlayer = new JTextField();
217             content.add(newPlayer, BorderLayout.NORTH);
218 
219             JButton newPlayerButton = new JButton("Create Player");
220             newPlayerButton.addActionListener(new AddPlayerListener(newPlayer, currentStandings));
221             content.add(newPlayerButton, BorderLayout.SOUTH);
222 
223             frame.setContentPane(content);
224             frame.pack();
225             frame.setLocationByPlatform(true);
226             frame.setVisible(true);
227 
228         }
229     }//GEN-LAST:event_addPlayerButtonActionPerformed

239     private void shoStandingsButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_shoStandingsButtonActionPerformed
240         if (this.currentStandings != null) {
241             JFrame frame = new JFrame("Standings ");
242             frame.setDefaultCloseOperation(HIDE_ON_CLOSE);
243             frame.setPreferredSize(new Dimension(800, 400));
244 
245             JPanel panel = new JPanel();
246 
247             panel.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
248             panel.setLayout(new GridLayout(getStandingsSize() + 1, 10));
249             panel.add(new JLabel(" Place "));
250             panel.add(new JLabel(" Name "));
251             panel.add(new JLabel(" Games Played "));
252             panel.add(new JLabel(" Wins "));
253             panel.add(new JLabel(" Loss "));
254             panel.add(new JLabel(" Ties "));
255             panel.add(new JLabel(" Goals For"));
256             panel.add(new JLabel(" Goals Against "));
257             panel.add(new JLabel(" Win% "));
258             panel.add(new JLabel(" Points "));
259             int index = 1;
260             for (Player player : currentStandings.getPlayers()) {
261                 panel.add(new JLabel(" " + index));
262                 panel.add(new JLabel(" " + player.getName()));
263                 panel.add(new JLabel(" " + player.getGamesPlayed()));
264                 panel.add(new JLabel(" " + player.getWins()));
265                 panel.add(new JLabel(" " + player.getLosses()));
266                 panel.add(new JLabel(" " + player.getTies()));
267                 panel.add(new JLabel(" " + player.getGoalsFor()));
268                 panel.add(new JLabel(" " + player.getGoalsAgainst()));
269                 panel.add(new JLabel(" " + player.getWinPrecentage()));
270                 panel.add(new JLabel(" " + player.getPoints()));
271                 index++;
272             }
273             frame.setContentPane(panel);
274             frame.pack();
275             frame.setLocationByPlatform(true);
276             frame.setVisible(true);
277 
278         }
279     }//GEN-LAST:event_shoStandingsButtonActionPerformed
