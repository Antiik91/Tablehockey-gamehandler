10  
11      /**
12       * Class is used to store Standings to be used in future.
13       *
14       * @author janantik
15       *
16       */
17      public DataStorage() {
18      }
19  
20      /**
21       * Method stores Standings to preselected folder in .ser.
22       *
23       * @param standings Standings to be stored.
24       */
25      public static void saveData(Standings standings) {
26          if (standings == null) {
27              throw new IllegalArgumentException("Standings can't be null!");
28          } else {
29              String standingsName = "./src/tmp/" + standings.getName() + ".ser";
30              try {
31                  FileOutputStream standingsOut = new FileOutputStream(standingsName);
32                  ObjectOutputStream out = new ObjectOutputStream(standingsOut);
33                  out.writeObject(standings);
34                  out.close();
35                  standingsOut.close();
36  //            System.out.println("The standings is saved in /tmp/"
37  //                    + standingsName + ".ser");
38              } catch (IOException e) {
39                  e.printStackTrace();
40              }
41          }
42      }
43  }
